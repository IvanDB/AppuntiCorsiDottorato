\section{Spazi di Sobolev $W^{1, p}$}
\stab
\ptab
\defg Dati $\Omega \subseteq \R^N$ e $p \in [1, +\infty]$\n
	\tdx $W^{1, p}(\Omega) = \left\{u \in L^p(\Omega) \bmid \exists g \in L^p\(\Omega, \R^N\) \tc \forall \varphi \in \Tsp(\Omega) \quad \intg_\Omega u \fraz{\derp \varphi}{\derp x_i} = -\intg_\Omega g_i\varphi\right\}$\n
	\notg $g_i = \fraz{\derp u}{\derp x_i}$, $g = \nabla u$\n
	\prop $\|\cdot\|_{W^{1, p}(\Omega)} =  \|\cdot\|_{L^p(\Omega)} + \|\nabla\cdot\|_{L^p(\Omega)}$ norma su $W^{1, p}(\Omega) \tc \(W^{1, p}(\Omega), \|\cdot\|_{W^{1, p}(\Omega)}\)$ spazio di Banach\n
\n

\defg Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$, $\{u_n\} \subseteq W^{1, p}(\Omega)$ e $u \in W^{1, p}(\Omega)$\n
	\tdx $u_n \convIn{W^{1, p}(\Omega)} u$ \t\t\t\t\t\t\t se \t $u_n, \nabla u_n \convIn{L^p(\Omega)} u, \nabla u$\n
\n
\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $\{u_n\} \subseteq W^{1, p}(\Omega) \tc \exists C > 0 \tc \forall n \quad \|u_n\|_{W^{1, p}(\Omega)} \leq C$\n
	\tdx $\bullet$ $p \in (1, +\infty) \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conWIn{L^p(\Omega)} u, \nabla u$\n
	\tdx $\bullet$ $p = +\infty \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conS u, \nabla u$\n
	\cor $p \in (1, +\infty) \imp u_{n_k} \convIn{L^p(\Omega)} u$\n
\n
\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato\n
	\tdx $\forall p \in [1, +\infty] \quad W^{1, p}(\Omega) = \overline{C^{\infty}\(\ov{\Omega}\)}^{\|\cdot\|_{W^{1, p}(\Omega)}}$\n
\n

\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $p \in [1, +\infty]$\n
	\tdx $\bullet$ $p \in [1, N) \imp \exists p* = \fraz{pN}{N-p} \tc \begin{cases} W^{1, p}(\Omega) \into L^{p*}(\Omega) \\ \forall q < p* \qquad W^{1, p}(\Omega) \intc L^{q}(\Omega) \end{cases}$\n
	\tdx $\bullet$ $p = N \imp \forall q < +\infty \quad W^{1, p}(\Omega) \intc L^{q}(\Omega)$\n
	\tdx $\bullet$ $p > N \imp \exists \alpha_p = \fraz{p-N}{p} \tc \begin{cases} W^{1, p}(\Omega) \into C^{0, \alpha_p}(\Omega) \\ \forall \beta < \alpha_p \qquad W^{1, p}(\Omega) \intc C^{0, \beta}(\Omega) \end{cases}$\n
\n

\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$ e $u \in W^{1, p}(\Omega)$\n
	\tdx $\forall F \in C^1(\R) \cap Lip(\R) \quad F \circ u \in W^{1, p}(\Omega) \tc \nabla(F \circ u) = F'(u) \nabla u$\n
\n
\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$\n
	\tdx $\forall u, v \in W^{1, p}(\Omega) \cap L^{\infty}(\Omega) \quad uv \in W^{1, p}(\Omega) \cap L^{\infty}(\Omega) \tc \nabla(uv) = v\nabla u  + u\nabla v$\n
\n

\section{Funzioni $BV$}
\stab
\ptab
\defg Dati $\Omega \subseteq \R^N$ Lipschitz limitato\n
	\tdx $BV(\Omega) = \left\{u \in L^1(\Omega) \bmid \exists \mu \in \mathcal{M}\(\Omega, \R^N\) \tc \forall \varphi \in \Tsp(\Omega) \quad \intg_\Omega u \fraz{\derp \varphi}{\derp x_i} = \intg_\Omega \varphi d\mu_i\right\}$\n
	\notg $\mu = Du$\n
	\prop $\|\cdot\|_{BV(\Omega)} =  \|\cdot\|_{L^(\Omega)} + \|D\cdot\|_{\mathcal{M}\(\Omega, \R^N\)}$ norma su $BV(\Omega) \tc \(BV(\Omega), \|\cdot\|_{BV(\Omega)}\)$ spazio di Banach\n
\n
\teog Dati $\Omega \subseteq \R^N$ Lipschitz limitato\n
	\tdx $\forall u \in BV(\Omega) \quad \exists \{u_n\}\subset C^{\infty}\(\ov{\Omega}\) \tc \begin{cases} u_n \convIn{L^1(\Omega)} u \qquad\qquad Du_n \conWIn{\mathcal{M}\(\Omega, \R^N\)} Du \\ \|Du_n\|_{\mathcal{M}\(\Omega, \R^N\)} \conv \|Du\|_{\mathcal{M}\(\Omega, \R^N\)}\end{cases}$\n
\n
\teog Dato $\Omega \subseteq \R^N$ Lipschitz limitato\n
	\tdx $BV(\Omega)$ immerge come $W^{1, 1}(\Omega)$\n
\n
\defg $\Sigma \subseteq \R^N$ $k$-rettificabile \t\t\t\t\t\t se \t $\exists \begin{cases} \Sigma_0 \subseteq \R^N \tc \mathcal{H}^k(\Sigma_0) = 0 \\ \{M_i \subseteq \R^N\}$ $C^1$ sottovarietÃ $\end{cases} \tc \Sigma \subseteq \Sigma_0 \cup \union_{i} M_i$\n

\teog Dato $\Sigma \subseteq \R^N$ $k$-rettificabile\n
	\tdx $\bullet$ $\mathcal{H}^k$-q.o. $\quad \exists T_x\Sigma$\n
