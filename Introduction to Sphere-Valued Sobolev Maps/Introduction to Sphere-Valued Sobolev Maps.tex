\subsection{Distribuzioni}
\notz{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\Tsp(\Omega) = C_c^{\infty}(\Omega)$}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $\{u_n\} \subseteq \Tsp(\Omega)$ e $u \in \Tsp(\Omega)$}
	{$u_n \convIn{\Tsp(\Omega)} u$ \defif{$\begin{cases} \exists K \subset \Omega$ compatto $\tc \forall n \quad \supp u_n \subseteq K \\ \forall \alpha \quad D^\alpha u_n \convU D^\alpha u\end{cases}$}}


\defc{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\dist(\Omega) = \{T : \Tsp(\Omega) \to \R \mid T$ lin e cont.$\}$ \defnm{spazio delle distribuzioni su $\Omega$}}
\notz{}{$\la T, u\ra = T(u)$}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $\{T_n\} \subseteq \dist(\Omega)$ e $T \in \dist(\Omega)$}
	{$T_n \conS T$ \defif{$\forall u \in \Tsp(\Omega) \quad \la T_n, u\ra \conv \la T, u\ra$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $\{T_n\} \subseteq \dist(\Omega)$}
	{$\forall u \in \Tsp(\Omega) \quad \sup_{n} \la T_n, u\ra < +\infty \imp \exists \begin{cases} \{T_{n_k}\} \subseteq \{T_n\} \\ T \in \dist(\Omega)\end{cases} \tc T_{n_k} \conS T$}
\edim

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $f \in L^1_{loc}(\Omega)$}
	{$T_f : \Tsp(\Omega) \to \R \tc \la T_f, u\ra = \intg_\Omega f(x)u(x)dx$}
	\prop{$T_f \in \dist(\Omega)$}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $T \in \dist(\Omega)$}
	{$\bullet$ $\fraz{\derp T}{\derp x_j} : \Tsp(\Omega)  \to \R \tc \la \fraz{\derp T}{\derp x_j}, u\ra = \la T, \fraz{\derp u}{\derp x_j}\ra$ \defnm{derivata $j$-esima di $T$}\n
	 $\bullet$ $\nabla T$, $\div T$, $\dots$}
	\prop{$\forall j \quad \fraz{\derp T}{\derp x_j}$, $\nabla T$, $\div T$, $\dots$ $\in \dist(\Omega)$}

\teoc{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\fraz{\derp}{\derp x_j} : \Tsp(\Omega) \to \Tsp(\Omega)$ lineare e continuo}
\edim

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $T \in \dist(\Omega)$ e $\rho \in \Tsp(\Omega)$}
	{$\bullet$ $\tilde \rho = \rho \circ \cdot_{-1}$\n
	 $\bullet$ $T * \rho : \Tsp(\Omega) \to \R \tc \la T * \rho, u \ra = \la T, \tilde\rho u \ra$}
	\prop{$T * \rho \in \dist(\Omega)$}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $T \in \dist(\Omega)$ e $\rho \in \Tsp(\Omega)$}
	{$\forall j \quad \fraz{\derp}{\derp x_j}(T * \rho) = \fraz{\derp T}{\derp x_j} * \rho = T * \fraz{\derp \rho}{\derp x_j}$}
\edim

\subsection{Spazi di Sobolev $W^{1, p}$}
\defc{Dati $\Omega \subseteq \R^N$ e $p \in [1, +\infty]$}
	{$W^{1, p}(\Omega) = \left\{u \in L^p(\Omega) \bmid \exists g \in L^p\(\Omega, \R^N\) \tc \forall \varphi \in \Tsp(\Omega) \quad \intg_\Omega u \fraz{\derp \varphi}{\derp x_i} = -\intg_\Omega g_i\varphi\right\}$}
	\notz{}{$g_i = \fraz{\derp u}{\derp x_i}$, $g = \nabla u$}
	\prop{$\|\cdot\|_{W^{1, p}(\Omega)} =  \|\cdot\|_{L^p(\Omega)} + \|\nabla\cdot\|_{L^p(\Omega)}$ norma su $W^{1, p}(\Omega) \tc \(W^{1, p}(\Omega), \|\cdot\|_{W^{1, p}(\Omega)}\)$ spazio di Banach}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$, $\{u_n\} \subseteq W^{1, p}(\Omega)$ e $u \in W^{1, p}(\Omega)$}
	{$\bullet$ $u_n \convIn{W^{1, p}(\Omega)} u$ \defif{$u_n, \nabla u_n \convIn{L^p(\Omega)} u, \nabla u$}\n
	$\bullet$ $u_n \conWIn{W^{1, p}(\Omega)} u$ \defif{$u_n, \nabla u_n \conWIn{L^p(\Omega)} u, \nabla u$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $\{u_n\} \subseteq W^{1, p}(\Omega) \tc \exists C > 0 \tc \forall n \quad \|u_n\|_{W^{1, p}(\Omega)} \leq C$}
	{$\bullet$ $p \in (1, +\infty) \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conWIn{L^p(\Omega)} u, \nabla u$\n
	 $\bullet$ $p = +\infty \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conS u, \nabla u$}
\edim

\corc{}{$p \in (1, +\infty) \imp u_{n_k} \convIn{L^p(\Omega)} u$}
\edim

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\forall p \in [1, +\infty] \quad W^{1, p}(\Omega) = \overline{C^{\infty}\(\ov{\Omega}\)}^{\|\cdot\|_{W^{1, p}(\Omega)}}$}
\edim

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $p \in [1, +\infty]$}
	{$\bullet$ $p \in [1, N) \imp \exists p^* = \fraz{pN}{N-p} \tc \begin{cases} W^{1, p}(\Omega) \into L^{p^*}(\Omega) \\ \forall q < p* \qquad W^{1, p}(\Omega) \intc L^{q}(\Omega) \end{cases}$\n
	 $\bullet$ $p = N \imp \forall q < +\infty \quad W^{1, p}(\Omega) \intc L^{q}(\Omega)$\n
	 $\bullet$ $p > N \imp \exists \alpha_p = \fraz{p-N}{p} \tc \begin{cases} W^{1, p}(\Omega) \into C^{0, \alpha_p}(\Omega) \\ \forall \beta < \alpha_p \qquad W^{1, p}(\Omega) \intc C^{0, \beta}(\Omega) \end{cases}$}
\edim

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$ e $u \in W^{1, p}(\Omega)$}
	{$\forall F \in C^1(\R) \cap Lip(\R) \quad F \circ u \in W^{1, p}(\Omega) \tc \nabla(F \circ u) = F'(u) \nabla u$}

\subsection{Funzioni $BV$}

	{$BV(\Omega) = \left\{u \in L^1(\Omega) \bmid \exists \mu \in \M\(\Omega, \R^N\) \tc \forall \varphi \in \Tsp(\Omega) \quad \intg_\Omega u \fraz{\derp \varphi}{\derp x_i} = \intg_\Omega \varphi d\mu_i\right\}$}
	\notz{}{$\mu = Du$}
	\prop{$\|\cdot\|_{BV(\Omega)} =  \|\cdot\|_{L^1(\Omega)} + \|D\cdot\|_{\mathcal{M}\(\Omega, \R^N\)}$ norma su $BV(\Omega) \tc \(BV(\Omega), \|\cdot\|_{BV(\Omega)}\)$ spazio di Banach}

\edim

\lemc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $p \in [1, +\infty]$}
	{$\forall \varphi \in u \in W^{1, p}\(\Omega, \R\) \quad e^{i\varphi} = \(\cos \varphi, \sin\varphi\) \in W^{1, p}\(\Omega, \S^1\)$}
\edim 

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$ e $u \in W^{1, p}\(\Omega, \S^1\)$}
	{$\varphi \in W^{1, p}\(\Omega, \R\) \tc u = e^{i\varphi}$ \defnm{lifting di $u$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato semplicemente connesso e $u : \Omega \to \S^1$}
	{$u \in C^k\(\Omega, \S^1\) \imp \exists \varphi \in C^k(\Omega, \R) \tc \forall x \in \Omega \quad u(x) = e^{i\varphi(x)}$}

\dimc{Dato $\pi(t) = e^{it}$ il rivestimento universale di $\S^1$, poichè $\Omega$ è semplicemente connesso, si ha \n
		$\forall u \in C^0\(\Omega, \S^1\) \quad \exists \varphi \in C^0(\Omega, \R) \tc u = \pi \circ \varphi.$ \n
		Inoltre poichè $\pi$ è localmente invertibile con inversa $C^{\infty}$ e $u$ è locamente non-suriettiva si ha \n
		$\forall x_0 \in \Omega \qquad \varphi|_{B_r(x_0)} = \pi^{-1}|_{u\(B_r(x_0)\)} \circ u|_{B_r(x_0)} \quad$ ovvero $\quad \varphi \in C^k(\Omega, \R).$}

\teoc{$\exists \Omega \subseteq \R^N$ Lipschitz limitato non semplicemente connesso $\tc \exists u \in C^\infty(\Omega, \S^1) \tc \nexists \varphi \in C^0(\Omega) \tc u = e^{i\varphi}$}{}
\dimc{Presi $\Omega = B_2 \dif B_1 \subset \R^2$ e $u(x) =\frac{x}{|x|}$ per assurdo si avrebbe omotopia tra $\Id_{\S^1}$ e una costante.}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $u \in W^{1, p}\(\Omega, \R^2\)$}
	{$\j u = u_1\nabla u_2 - u_2\nabla u_1$}
\prop{$\forall u \in W^{1, 1}\(\Omega, \S^1\) \tc \exists \varphi \in W^{1, 1}\(\Omega, \S^1\)$ lifting di $u \qquad \j u = \nabla \varphi$}

\lemc{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\forall u \in W^{1,1}(\Omega, \S^1) \quad \j u = \nabla u$}
	\tdim

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato semplicemente connesso e $V \in W^{1, p}\(\Omega, \R^N\)$}
	{$\curl(V) \in \R^{N\times N}_{\text{skew-sym}} \tc \(\curl(V)\)_{ij} = \fraz{\derp V_i}{\derp x_j} - \fraz{\derp V_j}{\derp x_i}$}
\prop{$\forall u \in W^{1, 1}\(\Omega, \R^2\) \tc \j u \in C^1(\Omega, \R^2) \quad \curl(\j u) = 0$}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $u \in W^{1, p}\(\Omega, \R^2\)$}
	{$\J u = \fraz{1}{2}\curl(\j u) \in \dist\(\Omega, \R^{N\times N}_{\text{skew-sym}}\) $ \defnm{Jacobiano distribuzionale}}
\prop{$u \in W^{1,2}(\Omega, \R^2) \quad \J u = \det(\nabla u)$}
\prop{$N = 2 \imp \forall \Psi \in \Tsp(\Omega, \R) \quad \la \J u, \Psi\ra = -\frac{1}{2}\la \j u, \nabla^{\perp}\Psi\ra$}
\prop{$N = 3 \imp \forall \Psi \in \Tsp(\Omega, \R^3) \quad \la \J u, \Psi\ra = \la \j u, \curl(\Psi)\ra$}

\lemc[1][1]{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $u \in W^{1, 1}\(\Omega, \S^1\)$}
	{$\forall j \quad u \perp \fraz{\derp u}{\derp x_j}$}
\dimc{$u \in W^{1, 1}\(\Omega, \S^1\) \imp u_1^2 + u_2^2 = 1 \imp \forall j \quad \fraz{\derp}{\derp x_j}\left[u_1^2 + u_2^2\right] = 2u_1\fraz{\derp u_1}{\derp x_j} + 2u_2\fraz{\derp u_2}{\derp x_j} = 0$\n
	$\imp \la u, \fraz{\derp u}{\derp x_j}\ra_{\R^2} = 0 \imp u \perp \fraz{\derp u}{\derp x_j}$}

\lemc[2][2]{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $u \in W^{1, 1}\(\Omega, \S^1\)$}
	{$\j u \equiv 0 \iff \nabla u \equiv 0 \iff u$ costante}
\dimc{$\j u \equiv 0 \imp  u_1\fraz{\derp u_2}{\derp x_j} + u_2\fraz{\derp u_1}{\derp x_j} = 0 \imp u_1^2\fraz{\derp u_2}{\derp x_j} + u_1u_2\fraz{\derp u_1}{\derp x_j} = 0 \impref[lem]{1} u_1^2\fraz{\derp u_1}{\derp x_j} - u_2^2\fraz{\derp u_2}{\derp x_j} = 0$}

\lemc[3][3]{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}
	{$\forall u, v \in W^{1, p}(\Omega, \S^1) \quad \j(u \cdot v) = \j u + \j v$}
