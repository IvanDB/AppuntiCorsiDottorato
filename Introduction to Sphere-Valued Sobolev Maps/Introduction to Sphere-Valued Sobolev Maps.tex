\defc{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}

\notz{}{$\la T, u\ra = T(u)$}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $\{T_n\} \subseteq \dist(\Omega)$ e $T \in \dist(\Omega)$}
	{$T_n \conS T$ \defif{$\forall u \in \Tsp(\Omega) \quad \la T_n, u\ra \conv \la T, u\ra$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $\{T_n\} \subseteq \dist(\Omega)$}
	{$\forall u \in \Tsp(\Omega) \quad \sup_{n} \la T_n, u\ra < +\infty \imp \exists \begin{cases} \{T_{n_k}\} \subseteq \{T_n\} \\ T \in \dist(\Omega)\end{cases} \tc T_{n_k} \conS T$}
\edim

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $f \in L^1_{loc}(\Omega)$}

	 $\bullet$ $\nabla T$, $\div T$, $\dots$}
	\prop{$\forall j \quad \fraz{\derp T}{\derp x_j}$, $\nabla T$, $\div T$, $\dots$ $\in \dist(\Omega)$}

\teoc{Dato $\Omega \subseteq \R^N$ Lipschitz limitato}

\edim

\subsection{Spazi di Sobolev $W^{1, p}$}
\defc{Dati $\Omega \subseteq \R^N$ e $p \in [1, +\infty]$}
	{$W^{1, p}(\Omega) = \left\{u \in L^p(\Omega) \bmid \exists g \in L^p\(\Omega, \R^N\) \tc \forall \varphi \in \Tsp(\Omega) \quad \intg_\Omega u \fraz{\derp \varphi}{\derp x_i} = -\intg_\Omega g_i\varphi\right\}$}
	\notz{}{$g_i = \fraz{\derp u}{\derp x_i}$, $g = \nabla u$}
	\prop{$\|\cdot\|_{W^{1, p}(\Omega)} =  \|\cdot\|_{L^p(\Omega)} + \|\nabla\cdot\|_{L^p(\Omega)}$ norma su $W^{1, p}(\Omega) \tc \(W^{1, p}(\Omega), \|\cdot\|_{W^{1, p}(\Omega)}\)$ spazio di Banach}

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$, $\{u_n\} \subseteq W^{1, p}(\Omega)$ e $u \in W^{1, p}(\Omega)$}
	{$\bullet$ $u_n \convIn{W^{1, p}(\Omega)} u$ \defif{$u_n, \nabla u_n \convIn{L^p(\Omega)} u, \nabla u$}\n
	$\bullet$ $u_n \conWIn{W^{1, p}(\Omega)} u$ \defif{$u_n, \nabla u_n \conWIn{L^p(\Omega)} u, \nabla u$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato e $\{u_n\} \subseteq W^{1, p}(\Omega) \tc \exists C > 0 \tc \forall n \quad \|u_n\|_{W^{1, p}(\Omega)} \leq C$}
	{$\bullet$ $p \in (1, +\infty) \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conWIn{L^p(\Omega)} u, \nabla u$\n
	 $\bullet$ $p = +\infty \imp \exists \{u_{n_k}\} \subseteq \{u_n\}, u \in W^{1, p}(\Omega) \tc u_{n_k}, \nabla u_{n_k} \conS u, \nabla u$}
\edim

\corc{}{$p \in (1, +\infty) \imp u_{n_k} \convIn{L^p(\Omega)} u$}
\edim

\defc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato, $p \in [1, +\infty]$ e $u \in W^{1, p}\(\Omega, \S^1\)$}
	{$\varphi \in W^{1, p}\(\Omega, \R\) \tc u = e^{i\varphi}$ \defnm{lifting di $u$}}

\teoc{Dati $\Omega \subseteq \R^N$ Lipschitz limitato semplicemente connesso e $u : \Omega \to \S^1$}
	{$u \in C^k\(\Omega, \S^1\) \imp \exists \varphi \in C^k(\Omega, \R) \tc \forall x \in \Omega \quad u(x) = e^{i\varphi(x)}$}

\dimc{Dato $\pi(t) = e^{it}$ il rivestimento universale di $\S^1$, poichè $\Omega$ è semplicemente connesso, si ha \n
		$\forall u \in C^0\(\Omega, \S^1\) \quad \exists \varphi \in C^0(\Omega, \R) \tc u = \pi \circ \varphi.$ \n
		Inoltre poichè $\pi$ è localmente invertibile con inversa $C^{\infty}$ e $u$ è locamente non-suriettiva si ha \n
		$\forall x_0 \in \Omega \qquad \varphi|_{B_r(x_0)} = \pi^{-1}|_{u\(B_r(x_0)\)} \circ u|_{B_r(x_0)} \quad$ ovvero $\quad \varphi \in C^k(\Omega, \R).$}

\teoc{$\exists \Omega \subseteq \R^N$ Lipschitz limitato non semplicemente connesso $\tc \exists u \in C^\infty(\Omega, \S^1) \tc \nexists \varphi \in C^0(\Omega) \tc u = e^{i\varphi}$}{}
\dimc{Presi $\Omega = B_2 \dif B_1 \subset \R^2$ e $u(x) =\frac{x}{|x|}$ per assurdo si avrebbe omotopia tra $\Id_{\S^1}$ e una costante.}

\section{Problemi di minimo con condizioni al bordo}

\subsection{Spazi di Sobolev frazionari}
\defc{Dati $\Omega \subseteq \R^N$ aperto Lipschitz, $s \in (0, 1)$ e $p \geq 1$}
	{$\bullet$ $[\;\cdot\;]_{s, p} = \(\iintg_{\Omega\times\Omega} \fraz{|{\;\cdot\;}(y) - {\;\cdot\;}(x)|^p}{|y-x|^{N+sp}}dxdy\)^{\frac{1}{p}}$ \\
	 $\bullet$ $W^{s, p}(\Omega) = \left\{u \in L^p(\Omega) \mid [u]_{s, p}^p < +\infty\right\}$ \\ 
	 $\bullet$ $\|\cdot\|_{W^{s, p}(\Omega)} = \(\|\cdot\|_{L^p(\Omega)} + [\;\cdot\;]_{s, p}\)^{\frac{1}{p}}$}
	\prop{$\(W^{s, p}(\Omega), \|\cdot\|_{W^{s, p}(\Omega)}\)$ spazio di Banach}

\prpc{Dato $\Omega \subseteq \R^N$ Lipschitz}
	{$\bullet$ $\forall p\geq 1 \quad \forall s \in (0, 1) \quad Lip(\Omega) \subseteq W^{s, p}(\Omega) \subseteq L^p(\Omega)$ \\ 
	 $\bullet$ $\forall p \geq 1\quad \forall s_1, s_2 \in (0, 1) \quad s_1 < s_2 \imp W^{s_2, p}(\Omega) \subseteq W^{s_1, p}(\Omega)$}
	\edim

\teoc[BBM formula, '01]{Dato $\Omega \subseteq \R^N$ aperto Lipschitz} 
	{$\forall p \geq 1 \quad \exists K_{N, p} > 0 \tc \limf_{s \to 1^-} (1-s)[u]_{s, p} = \begin{cases} K_{N, p} \|\nabla u\|_{L^p(\Omega)}^p &$se $p>1$ e $u \in W^{1, p}(\Omega) \\ K_{N, p} \|Du\|_{\mathcal{M}(\Omega)}&$se $p=1$ e $u \in BV(\Omega) \\ +\infty&$altrimenti$\end{cases}$}
	\edim
\teoc[Ponse, '04]{BBM formula vale anche per $\Glim$}
	\edim

\teoc[Mazya - Shaposhikova, '02]{Dato $\Omega \subseteq \R^N$ aperto Lipschitz} 
	{$\forall p \geq 1 \quad \exists K'_{N, p} > 0 \tc \forall u \in \union_{s > 0} W^{s, p}(\Omega) \quad \limf_{s \to 0} s[u]_{s, p} = K'_{N, p}\|u\|_{L^p(\Omega)}^p$}
	\edim

\subsection{Operatore di traccia}
\teoc[Trace theorem]{Dati $\Omega \subseteq \R^N$ aperto Lipschitz e $p \in [1, +\infty)$} 
	{$\bullet$ $\exists T : W^{1, p}(\Omega) \to W^{1 - \frac{1}{p}, p}(\derp\Omega)$ lineare limitato $\tc \forall u \in W^{1, p}(\Omega) \cap C^0\(\Omega\) \quad Tu = u|_{\derp\Omega}$ \\
	 $\bullet$ $\exists C_p > 0 \tc \forall g \in W^{1 - \frac{1}{p}, p}(\derp\Omega) \quad \exists v \in u \in W^{1, p}(\Omega) \cap C^\infty\(\Omega\) \tc \begin{cases} Tv = g \\ \|\nabla v\|_{L^p(\Omega)} \leq C_p [g]_{1 - \frac{1}{p}, p}\end{cases}$}
	\edim

\teoc[Gagliardo - Niremberg inequality]{Dato $\Omega \subseteq \R^N$ aperto Lipschitz}
	{$\forall p \quad \exists C_p > 0 \tc \forall u \in BV(\Omega) \cap L^\infty(\Omega) \quad \forall \theta \in (0, 1) \quad [u]_{\frac{\theta}{p}, p} \leq C_p\|Du\|_{\mathcal{M}(\Omega)}^{\frac{1}{p}}\|u\|_{L^{\infty}(\Omega)}^p$}
	\edim
\corc{$p \in (1, 2) \imp BV(\Omega) \cap L^\infty(\Omega) \into W^{1 - \frac{1}{p}, p}(\derp\Omega)$}
	\edim

\subsection{Problemi di Dirichlet}
\notz {Dato $\Omega \subseteq \R^N$}
	{$\Omega$ \tempReg \qquad \text{se} \qquad $\Omega \isom B_1$ aperto liscio}

\defc {Dati $\Omega \subseteq \R^N$ \tempReg e $g \in C^\infty(\derp\Omega)$}
	{$W^{1, p}_g(\Omega, \S^1) = \{u \in W^{1, p}(\Omega) \mid Tu = g\}$}

\defc {Dati $\Omega \subseteq \R^N$ \tempReg e $g \in C^\infty(\derp\Omega, \S^1)$}
	{$\bullet$ $W_p(g) = \inf\left\{\intg_\Omega |\nabla u|^p \bmid u \in W^{1, p}_g(\Omega, \S^1)\right\}$ \\ 
	 $\bullet$ $J_p(g) = \inf\left\{\intg_\Omega |\nabla u|^p \bmid u \in W^{1, p}_g(\Omega, \S^1) \tc \J u = 0\right\}$ \\ 
	 $\bullet$ $C_p(g) = \inf\left\{\intg_\Omega |\nabla u|^p \bmid u \in W^{1, p}_g(\Omega, \S^1) \cap C^0(\ov{\Omega}, \S^1)\right\}$}
	\prop{$\forall p \quad W_p(g) \leq J_p(g) \leq C_p(g)$}

\teoc{Dati $N\geq 2$, $\Omega \subseteq \R^N$ \tempReg e $g \in C^\infty(\derp\Omega, \S^1) \tc \deg(g) =0$ se $N = 2$}
	{$\forall p \geq 2 \quad \exists \ov{u} \in C^{1, \alpha}(\ov{\Omega})$ minimizzatore per $W_p(g)$}
	\tdim
\corc{$\forall p \geq 2 \quad W_p(g) = J_p(g) = C_p(g)$}
	\edim

\prpc{Dati $N\geq 2$, $\Omega \subseteq \R^N$ \tempReg e $g \in C^\infty(\derp\Omega, \S^1)$}
	{$\bullet$ $\forall p \in (1, 2) \quad \forall X = W, J \quad \exists u^X_{p, g}$ minimizzatore per $X_p(g)$ \\
	 $\bullet$ $\forall p \in (1, 2) \quad \exists u^C_{p, g}$ minimizzatore per $C_p(g)$ dato $\deg(g) =0$ se $N = 2$}
	\tdim

\prpc{Dati $N\geq 2$, $\Omega \subseteq \R^N$ \tempReg, $g \in C^\infty(\derp\Omega, \S^1)$ e $p \in (1, 2)$}
	{$u^J_{p, g} = e^{i\varphi_J} \tc \varphi_j \in Sol\(\Delta_p \varphi = 0\)$}
	\edim
\corc{$u^J_{p, g} \in C^{1, \alpha}(\Omega, \S^1)$}{}
	\edim

\defc{Dati $\Omega \subseteq \R^N$ \tempReg e $u \in W^{1, p}(\Omega, \S^1)$}
	{$\bullet$ $u$ mappa $p$-armonica $\quad$ se $\quad$ $u \in Sol\(\Delta_p u = |\nabla u|^p u\)$ \\
	 $\bullet$ $u$ mappa $p$-armonica stazionaria $\qquad$ se $\quad$ $u$ $p$-armonica $\tc \forall \{\Phi_t\}_t \tc \Phi_0 = Id \quad \fraz{d}{dt}\Big|_{t = 0} \intg_\Omega |\nabla u \circ \Phi_t|^p = 0$}

\prpc{Dati $N\geq 2$, $\Omega \subseteq \R^N$ \tempReg, $g \in C^\infty(\derp\Omega, \S^1)$ e $p \in (1, 2)$}
	{$u^W_{p, g}$ mappa $p$-armonica stazionaria}
	\edim 
\corc{$\exists K \subset\subset \Omega$ compatto $\tc \mathcal{H}^{N-2}(K) < +\infty \tc u^W_{p, g} \in C^{1, \alpha}(\ov{\Omega} \dif K, \S^1)$}{}
	\edim
